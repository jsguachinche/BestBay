  $('#searchvalue').hide();
  $('#movil').click(function () {
    $('#searchvalue').show('slow');
      if (!swC) {
          $('.myCollapse').hide();
          swC = true;
          return;
      }
      if (swC) {
          $('.myCollapse').empty();
